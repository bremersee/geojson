118   /**
119    * Checks whether two geometry objects are equal.
120    *
121    * <p>Because the {@link GeometryCollection#equals(Geometry)} method throws an exception, this
122    * method is used in the GeoJSON classes.
123    *
124    * @param g1 one geometry
125    * @param g2 another geometry
126    * @return {@code true} if the geometries are equal otherwise {@code false}
127    */
128   public static boolean equals(final Geometry g1, final Geometry g2) {
129     if (g1 == null && g2 == null) {
130       return true;
131     }
132     if (g1 == null || g2 == null) {
133       return false;
134     }
135     if (g1 == g2) {
136       return true;
137     }
138     if (g1 instanceof GeometryCollection && g2 instanceof GeometryCollection) {
139       return equals((GeometryCollection) g1, (GeometryCollection) g2);
140     }
141     return g1.equals(g2);
142   }

144   /**
145    * Checks whether two geometry collections are equal.
146    *
147    * @param gc1 one geometry collection
148    * @param gc2 another geometry collection
149    * @return {@code true} if the geometry collections are equal otherwise {@code false}
150    */
151   private static boolean equals(final GeometryCollection gc1, final GeometryCollection gc2) {
152     if (gc1.getNumGeometries() != gc2.getNumGeometries()) {
153       return false;
154     }
155     for (int i = 0; i < gc1.getNumGeometries(); i++) {
156       Geometry g1 = gc1.getGeometryN(i);
157       Geometry g2 = gc2.getGeometryN(i);
158       if (!equals(g1, g2)) {
159         return false;
160       }
161     }
162     return true;
163   }

165   /**
166    * Calculate the bounding box of the specified geometry (see
167    * <a href="https://tools.ietf.org/html/rfc7946#section-5">bounding-boxes</a>).
168    *
169    * <p>A GeoJSON object MAY have a member named "bbox" to include information on the coordinate
170    * range for its Geometries, Features, or FeatureCollections.  The value of the bbox member MUST
171    * be an array of length 2*n where n is the number of dimensions represented in the contained
172    * geometries, with all axes of the most southwesterly point followed by all axes of the more
173    * northeasterly point. The axes order of a bbox follows the axes order of geometries.
174    *
175    * @param geometry the geometry
176    * @return {@code null} if the bounding box can not be calculated, otherwise the bounding box
177    */
178   public static double[] getBoundingBox(final Geometry geometry) {
179     return Optional.ofNullable(geometry)
180         .map(g -> getBoundingBox(Collections.singletonList(g)))
181         .orElse(null);
182   }

184   /**
185    * Calculate the bounding box of the specified geometries.
186    *
187    * @param geometries the geometries
188    * @return {@code null} if the bounding box can not be calculated, otherwise the bounding box
189    */
190   public static double[] getBoundingBox(final Collection<? extends Geometry> geometries) {
191     if (geometries == null || geometries.isEmpty()) {
192       return null;
193     }
194     double minX = Double.NaN;
195     double minY = Double.NaN;
196     double minZ = Double.NaN;
197     double maxX = Double.NaN;
198     double maxY = Double.NaN;
199     double maxZ = Double.NaN;
200     for (final Geometry geometry : geometries) {
201       if (geometry != null && geometry.getCoordinates() != null) {
202         final Coordinate[] coords = geometry.getCoordinates();
203         //noinspection ForLoopReplaceableByForEach
204         for (int i = 0; i < coords.length; i++) {
205           if (Double.isNaN(minX)) {
206             minX = coords[i].getX();
207           } else if (!Double.isNaN(coords[i].getX())) {
208             minX = Math.min(minX, coords[i].getX());
209           }
210           if (Double.isNaN(minY)) {
211             minY = coords[i].getY();
212           } else if (!Double.isNaN(coords[i].getY())) {
213             minY = Math.min(minY, coords[i].getY());
214           }
215           if (Double.isNaN(minZ)) {
216             minZ = coords[i].getZ();
217           } else if (!Double.isNaN(coords[i].getZ())) {
218             minZ = Math.min(minZ, coords[i].getZ());
219           }
220 
221           if (Double.isNaN(maxX)) {
222             maxX = coords[i].getX();
223           } else if (!Double.isNaN(coords[i].getX())) {
224             maxX = Math.max(maxX, coords[i].getX());
225           }
226           if (Double.isNaN(maxY)) {
227             maxY = coords[i].getY();
228           } else if (!Double.isNaN(coords[i].getY())) {
229             maxY = Math.max(maxY, coords[i].getY());
230           }
231           if (Double.isNaN(maxZ)) {
232             maxZ = coords[i].getZ();
233           } else if (!Double.isNaN(coords[i].getZ())) {
234             maxZ = Math.max(maxZ, coords[i].getZ());
235           }
236         }
237       }
238     }
239     if (!Double.isNaN(minX) && !Double.isNaN(maxX) && !Double.isNaN(minY) && !Double.isNaN(maxY)) {
240       if (!Double.isNaN(minZ) && !Double.isNaN(maxZ)) {
241         return new double[]{minX, minY, minZ, maxX, maxY, maxZ};
242       }
243       return new double[]{minX, minY, maxX, maxY};
244     }
245     return null;
246   }

261   /**
262    * Returns the coordinate in the north-west.
263    *
264    * @param boundingBox the bounding box
265    * @return the coordinate in the north-west
266    */
267   public static Coordinate getNorthWest(final double[] boundingBox) {
268     if (boundingBox == null || !(boundingBox.length == 4 || boundingBox.length == 6)) {
269       return null;
270     }
271     if (boundingBox.length == 6) {
272       // 0   1   2   3   4   5
273       // x0, y0, z0, x1, y1, z1
274       return createCoordinate(boundingBox[0], boundingBox[4]);
275     } else {
276       // 0   1   2   3
277       // x0, y0, x1, y1
278       return createCoordinate(boundingBox[0], boundingBox[3]);
279     }
280   }

334   /**
335    * Returns a polygon from the bounding box.
336    *
337    * @param boundingBox the bounding bos
338    * @param geometryFactory the geometry factory
339    * @return the polygon or {@code null} if the bounding box is {@code null} or empty
340    */
341   public static Polygon getBoundingBoxAsPolygon2D(
342       final double[] boundingBox,
343       final GeometryFactory geometryFactory) {
344 
345     final Coordinate sw = getSouthWest(boundingBox);
346     final Coordinate se = getSouthEast(boundingBox);
347     final Coordinate ne = getNorthEast(boundingBox);
348     final Coordinate nw = getNorthWest(boundingBox);
349     if (sw == null || se == null || ne == null || nw == null) {
350       return null;
351     }
352     float x1 = (float) sw.getX();
353     float x2 = (float) se.getX();
354     if (x1 == x2) {
355       return null;
356     }
357     float y1 = (float) sw.getY();
358     float y2 = (float) nw.getY();
359     if (y1 == y2) {
360       return null;
361     }
362     return createPolygon(
363         createLinearRing(Arrays.asList(sw, se, ne, nw, sw), geometryFactory),
364         geometryFactory);
365   }

529   public static Coordinate createCoordinate(final BigDecimal x, final BigDecimal y) {
530

532         .map(xx -> Optional.ofNullable(y)
533             .map(yy -> new Coordinate(xx.doubleValue(), yy.doubleValue()))
534             .orElseThrow(() -> new IllegalArgumentException("Y must not be null.")))
535         .orElseThrow(() -> new IllegalArgumentException("X must not be null."));
536   }

972    * Copy and apply filters.
973    *
974    * @param geometry the geometry
975    * @param filters the filters
976    * @return the copied and filtered geometry
977    */
978   public static Geometry copyAndApplyFilters(
979       final Geometry geometry,
980       final CoordinateFilter... filters) {
981     return Optional.ofNullable(geometry)
982         .map(g -> {
983           final Geometry result = g.copy();
984           if (filters != null) {
985             for (CoordinateFilter filter : filters) {
986               result.apply(filter);
987             }
988           }
989           return result;
990         })
991         .orElse(null);
992   }
