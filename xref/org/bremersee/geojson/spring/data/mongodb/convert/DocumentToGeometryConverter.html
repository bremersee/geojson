17  package org.bremersee.geojson.spring.data.mongodb.convert;
18  
19  import java.util.List;
20  import org.bson.Document;
21  import org.locationtech.jts.geom.Geometry;
22  import org.locationtech.jts.geom.GeometryFactory;
23  import org.springframework.data.convert.ReadingConverter;
24  
25  /**
26   * The document to geometry converter.
27   *
28   * @author Christian Bremer
29   */
30  @ReadingConverter
31  class DocumentToGeometryConverter extends AbstractDocumentToGeometryConverter<Geometry> {
32  
33    /**
34     * Instantiates a new document to geometry converter.
35     */
36    DocumentToGeometryConverter() {
37      super(null);
38    }
39  
40    /**
41     * Instantiates a new document to geometry converter.
42     *
43     * @param geometryFactory the geometry factory
44     */
45    DocumentToGeometryConverter(final GeometryFactory geometryFactory) {
46      super(geometryFactory);
47    }
48  
49    @Override
50    Geometry doConvert(final Document document) {
51      final String type = document.getString("type");
52  
53      if ("GeometryCollection".equals(type)) {
54        return new DocumentToGeometryCollectionConverter(getConvertHelper().getGeometryFactory())
55            .doConvert(document);
56      }
57  
58      //noinspection unchecked
59      final List<Object> coordinates = (List<Object>) document.get("coordinates");
60  
61      if ("Point".equals(type)) {
62        return getConvertHelper().createPoint(coordinates);
63      }
64      if ("LineString".equals(type)) {
65        return getConvertHelper().createLineString(coordinates);
66      }
67      if ("Polygon".equals(type)) {
68        return getConvertHelper().createPolygon(coordinates);
69      }
70      if ("MultiPoint".equals(type)) {
71        return getConvertHelper().createMultiPoint(coordinates);
72      }
73      if ("MultiLineString".equals(type)) {
74        return getConvertHelper().createMultiLineString(coordinates);
75      }
76      if ("MultiPolygon".equals(type)) {
77        return getConvertHelper().createMultiPolygon(coordinates);
78      }
79      throw new IllegalArgumentException("Geometry type [" + type + "] is unsupported.");
80    }
81  
82  }
