17  package org.bremersee.geojson.spring.boot.autoconfigure;
18  
19  import lombok.extern.slf4j.Slf4j;
20  import org.bremersee.geojson.GeoJsonGeometryFactory;
21  import org.springframework.boot.autoconfigure.AutoConfiguration;
22  import org.springframework.boot.autoconfigure.condition.ConditionalOnClass;
23  import org.springframework.boot.autoconfigure.condition.ConditionalOnMissingBean;
24  import org.springframework.boot.context.event.ApplicationReadyEvent;
25  import org.springframework.context.annotation.Bean;
26  import org.springframework.context.event.EventListener;
27  import org.springframework.util.ClassUtils;
28  
29  /**
30   * The GeoJSON geometry factory autoconfiguration.
31   *
32   * @author Christian Bremer
33   */
34  @ConditionalOnClass(name = {"org.bremersee.geojson.GeoJsonGeometryFactory"})
35  @AutoConfiguration
36  @Slf4j
37  public class GeoJsonGeometryFactoryAutoConfiguration {
38  
39    /**
40     * Init.
41     */
42    @EventListener(ApplicationReadyEvent.class)
43    public void init() {
44      log.info("""
45  
46              *********************************************************************************
47              * {}
48              *********************************************************************************""",
49          ClassUtils.getUserClass(getClass()).getSimpleName());
50    }
51  
52    /**
53     * Creates the GeoJSON geometry factory.
54     *
55     * @return the gGeoJSON geometry factory
56     */
57    @ConditionalOnMissingBean
58    @Bean
59    public GeoJsonGeometryFactory geoJsonGeometryFactory() {
60      return new GeoJsonGeometryFactory();
61    }
62  
63  }
