40  /**
41   * <p>
42   * A Jackson serializer for a {@link Geometry}.
43   * </p>
44   * 
45   * @author Christian Bremer
46   */
47  public class GeometrySerializer extends StdSerializer<Geometry> implements Serializable {
48  
49      private static final long serialVersionUID = 1L;
50  
51      private static final String TYPE_ATTRIBUTE_NAME = "type";
52  
53      private static final String COORDINATES_ATTRIBUTE_NAME = "coordinates";
54  
55      private static final String GEOMETRIES_ATTRIBUTE_NAME = "geometries";
56  
57      private static final NumberFormat NUMBER_FORMAT = NumberFormat.getInstance(Locale.US);
58  
59      static {
60          NUMBER_FORMAT.setMaximumFractionDigits(9);
61          NUMBER_FORMAT.setMaximumIntegerDigits(17);
62          NUMBER_FORMAT.setRoundingMode(RoundingMode.HALF_UP);
63          NUMBER_FORMAT.setGroupingUsed(false);
64      }
65  
66  
67      /**
68       * Default constructor.
69       */
70      public GeometrySerializer() {
71          super(Geometry.class, false);
72      }
73  
74      //@formatter:off
75      /* (non-Javadoc)
76       * @see com.fasterxml.jackson.databind.ser.std.StdSerializer#serialize(java.lang.Object, com.fasterxml.jackson.core.JsonGenerator, com.fasterxml.jackson.databind.SerializerProvider)
77       */
78      //@formatter:on
79      @Override
80      public void serialize(Geometry value, JsonGenerator jgen, SerializerProvider provider)
81              throws IOException {
82  
83          if (value == null) {
84              jgen.writeNull();
85          } else {
86              jgen.writeObject(create(value));
87          }
88      }
89  
90      private Map<String, Object> create(Geometry geometry) {
91          if (geometry instanceof Point) {
92              return createPoint((Point) geometry);
93          }
94  
95          if (geometry instanceof LineString) {
96              return createLine((LineString) geometry);
97          }
98  
99          if (geometry instanceof Polygon) {
100             return createPolygon((Polygon) geometry);
101         }
102 
103         if (geometry instanceof MultiPoint) {
104             return createMultiPoint((MultiPoint) geometry);
105         }
106 
107         if (geometry instanceof MultiLineString) {
108             return createMultiLine((MultiLineString) geometry);
109         }
110 
111         if (geometry instanceof MultiPolygon) {
112             return createMultiPolygon((MultiPolygon) geometry);
113         }
114 
115         if (geometry instanceof GeometryCollection) {
116             return createGeometryCollection((GeometryCollection) geometry);
117         }
118 
119         throw new IllegalArgumentException("Geometry [" + geometry
120                 + "] is unsupported. It must be an instance of Point, LineString, Polygon, MultiPoint, MultiLineString, MultiPolygon or GeometryCollection.");
121     }
122 
123     private Map<String, Object> createPoint(Point point) {
124         LinkedHashMap<String, Object> map = new LinkedHashMap<>();
125 
126         map.put(TYPE_ATTRIBUTE_NAME, "Point");
127         map.put(COORDINATES_ATTRIBUTE_NAME, createCoordinates(point.getCoordinate()));
128         return map;
129     }
130 
131     private Map<String, Object> createLine(LineString line) {
132         LinkedHashMap<String, Object> map = new LinkedHashMap<>();
133 
134         map.put(TYPE_ATTRIBUTE_NAME, "LineString");
135         map.put(COORDINATES_ATTRIBUTE_NAME, createCoordinates(line.getCoordinateSequence()));
136         return map;
137     }
138 
139     private Map<String, Object> createPolygon(Polygon polygon) {
140         LinkedHashMap<String, Object> map = new LinkedHashMap<>();
141 
142         map.put(TYPE_ATTRIBUTE_NAME, "Polygon");
143         map.put(COORDINATES_ATTRIBUTE_NAME, createCoordinates(polygon));
144         return map;
145     }
146 
147     private Map<String, Object> createMultiPoint(MultiPoint multiPoint) {
148         LinkedHashMap<String, Object> map = new LinkedHashMap<>();
149 
150         map.put(TYPE_ATTRIBUTE_NAME, "MultiPoint");
151         map.put(COORDINATES_ATTRIBUTE_NAME, createCoordinates(multiPoint));
152         return map;
153     }
154 
155     private Map<String, Object> createMultiLine(MultiLineString multiLine) {
156         LinkedHashMap<String, Object> map = new LinkedHashMap<>();
157 
158         map.put(TYPE_ATTRIBUTE_NAME, "MultiLineString");
159         map.put(COORDINATES_ATTRIBUTE_NAME, createCoordinates(multiLine));
160         return map;
161     }
162 
163     private Map<String, Object> createMultiPolygon(MultiPolygon multiPolygon) {
164         LinkedHashMap<String, Object> map = new LinkedHashMap<>();
165 
166         map.put(TYPE_ATTRIBUTE_NAME, "MultiPolygon");
167         map.put(COORDINATES_ATTRIBUTE_NAME, createCoordinates(multiPolygon));
168         return map;
169     }
170 
171     private Map<String, Object> createGeometryCollection(GeometryCollection geometryColllection) {
172         LinkedHashMap<String, Object> map = new LinkedHashMap<>();
173 
174         List<Map<String, Object>> geometries = new ArrayList<>(geometryColllection.getNumGeometries());
175         for (int i = 0; i < geometryColllection.getNumGeometries(); i++) {
176             geometries.add(create(geometryColllection.getGeometryN(i)));
177         }
178 
179         map.put(TYPE_ATTRIBUTE_NAME, "GeometryCollection");
180         map.put(GEOMETRIES_ATTRIBUTE_NAME, geometries);
181         return map;
182     }
183 
184     private double round(double value) {
185         if (Double.isNaN(value)) {
186             return value;
187         }
188         final String strValue = NUMBER_FORMAT.format(value);
189         return new BigDecimal(strValue).doubleValue();
190     }
191 
192     private List<Object> createCoordinates(Coordinate coordinate) {
193         if (coordinate == null) {
194             return Collections.emptyList();
195         }
196         final List<Object> list = new ArrayList<>(3);
197 
198         final double dx = coordinate.x;
199         list.add(round(dx));
200 
201         final double dy = coordinate.y;
202         list.add(round(dy));
203 
204         if (!Double.isNaN(coordinate.z)) {
205             final double dz = coordinate.z;
206             list.add(round(dz));
207         }
208         return list;
209     }
210 
211     private List<List<Object>> createCoordinates(CoordinateSequence coordinateSequence) {
212         if (coordinateSequence == null || coordinateSequence.size() == 0) {
213             return Collections.emptyList();
214         }
215         List<List<Object>> list = new ArrayList<>(coordinateSequence.size());
216         for (int n = 0; n < coordinateSequence.size(); n++) {
217             list.add(createCoordinates(coordinateSequence.getCoordinate(n)));
218         }
219         return list;
220     }
221 
222     private List<List<List<Object>>> createCoordinates(Polygon polygon) {
223         List<List<List<Object>>> list = new ArrayList<>();
224         list.add(createCoordinates(polygon.getExteriorRing().getCoordinateSequence()));
225         for (int i = 0; i < polygon.getNumInteriorRing(); i++) {
226             list.add(createCoordinates(polygon.getInteriorRingN(i).getCoordinateSequence()));
227         }
228         return list;
229     }
230 
231     private List<Object> createCoordinates(GeometryCollection geometryCollection) {
232         List<Object> list = new ArrayList<>(geometryCollection.getNumGeometries());
233         for (int i = 0; i < geometryCollection.getNumGeometries(); i++) {
234             Geometry g = geometryCollection.getGeometryN(i);
235             if (g instanceof Polygon) {
236                 list.add(createCoordinates((Polygon) g));
237             } else if (g instanceof LineString) {
238                 list.add(createCoordinates(((LineString) g).getCoordinateSequence()));
239             } else if (g instanceof Point) {
240                 list.add(createCoordinates((g).getCoordinate()));
241             }
242         }
243         return list;
244     }
