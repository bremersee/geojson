40  /**
41   * <p>
42   * A Jackson deserializer for a {@link Geometry}.
43   * </p>
44   * 
45   * @author Christian Bremer
46   */
47  public class GeometryDeserializer extends StdDeserializer<Geometry> {
48  
49      private static final long serialVersionUID = 1L;
50  
51      private GeometryFactory geometryFactory = new GeometryFactory();
52  
53      /**
54       * Default constructor.
55       */
56      public GeometryDeserializer() {
57          super(Geometry.class);
58      }
59  
60      /**
61       * Constructs a deserializer that uses the specified geometry factory.
62       * 
63       * @param geometryFactory
64       *            the geometry factory
65       */
66      public GeometryDeserializer(GeometryFactory geometryFactory) {
67          this();
68          setGeometryFactory(geometryFactory);
69      }
70  
71      /**
72       * Gets the geometry factory.
73       * 
74       * @return the geometry factory
75       */
76      protected GeometryFactory getGeometryFactory() {
77          return geometryFactory;
78      }
79  
80      /**
81       * Returns the geometry factory.
82       * 
83       * @param geometryFactory
84       *            the geometry factory
85       */
86      public void setGeometryFactory(GeometryFactory geometryFactory) {
87          if (geometryFactory != null) {
88              this.geometryFactory = geometryFactory;
89          }
90      }

92      //@formatter:off
93      /* (non-Javadoc)
94       * @see com.fasterxml.jackson.databind.JsonDeserializer#deserialize(com.fasterxml.jackson.core.JsonParser, com.fasterxml.jackson.databind.DeserializationContext)
95       */
96      //@formatter:on
97      @Override
98      public Geometry deserialize(JsonParser jp, DeserializationContext ctxt)
99              throws IOException, JsonProcessingException {
100 
101         String type = null;
102         List<Object> coordinates = new ArrayList<Object>();
103         List<Geometry> geometries = new ArrayList<Geometry>();
104 
105         JsonToken currentToken = null;
106         while ((currentToken = jp.nextValue()) != null) {
107             if (JsonToken.VALUE_STRING.equals(currentToken)) {
108                 if ("type".equals(jp.getCurrentName())) {
109                     type = jp.getText();
110                 }
111             } else if (JsonToken.START_ARRAY.equals(currentToken)) {
112                 if ("coordinates".equals(jp.getCurrentName())) {
113                     parseCoordinates(0, coordinates, jp, ctxt);
114                 } else if ("geometries".equals(jp.getCurrentName())) {
115                     geometries.addAll(parseGeometries(jp, ctxt));
116                 }
117             } else if (JsonToken.END_OBJECT.equals(currentToken)) {
118                 break;
119             }
120         }
121 
122         if ("GeometryCollection".equals(type)) {
123             return createGeometryCollection(geometries);
124         } else {
125             return createGeometry(type, coordinates);
126         }
127     }
128 
129     private void parseCoordinates(int depth, List<Object> coordinates,
130             JsonParser jp, DeserializationContext ctxt)
131                     throws IOException, JsonProcessingException {
132 
133         JsonToken currentToken = null;
134         while ((currentToken = jp.nextValue()) != null) {
135             if (JsonToken.END_ARRAY.equals(currentToken)) {
136                 break;
137             } else if (JsonToken.START_ARRAY.equals(currentToken)) {
138                 List<Object> list = new ArrayList<Object>();
139                 parseCoordinates(depth + 1, list, jp, ctxt);
140                 coordinates.add(list);
141             } else if (JsonToken.VALUE_NUMBER_FLOAT.equals(currentToken)) {
142                 coordinates.add(jp.getDoubleValue());
143             } else if (JsonToken.VALUE_NUMBER_INT.equals(currentToken)) {
144                 coordinates.add(Long.valueOf(jp.getLongValue()).doubleValue());
145             }
146         }
147     }
148 
149     private Coordinate createCoordinate(List<Object> coordinates) {
150         Coordinate coordinate = new Coordinate();
151         coordinate.x = coordinates.size() > 0 ? (double) coordinates.get(0)
152                 : Double.NaN;
153         coordinate.y = coordinates.size() > 1 ? (double) coordinates.get(1)
154                 : Double.NaN;
155         coordinate.z = coordinates.size() > 2 ? (double) coordinates.get(2)
156                 : Double.NaN;
157         return coordinate;
158     }
159 
160     private Coordinate[] createCoordinates(List<Object> coordinates) {
161         Coordinate[] coords = new Coordinate[coordinates.size()];
162         int i = 0;
163         for (Object obj : coordinates) {
164             @SuppressWarnings("unchecked")
165             List<Object> list = (List<Object>) obj;
166             coords[i] = createCoordinate(list);
167             i++;
168         }
169         return coords;
170     }
171 
172     private Point createPoint(List<Object> coordinates) {
173         return getGeometryFactory().createPoint(createCoordinate(coordinates));
174     }
175 
176     private LineString createLineString(List<Object> coordinates) { // List<List<Double>>
177         return getGeometryFactory()
178                 .createLineString(createCoordinates(coordinates));
179     }
180 
181     private Polygon createPolygon(List<Object> coordinates) { // List<List<List<Double>>>
182         List<Coordinate[]> list = new ArrayList<Coordinate[]>();
183         for (Object obj : coordinates) {
184             @SuppressWarnings("unchecked")
185             List<Object> coords = (List<Object>) obj;
186             list.add(createCoordinates(coords));
187         }
188         if (list.size() == 1) {
189             return getGeometryFactory().createPolygon(list.get(0));
190         } else {
191             GeometryFactory gf = getGeometryFactory();
192             LinearRing[] holes = new LinearRing[list.size() - 1];
193             for (int i = 1; i < list.size(); i++) {
194                 holes[i - 1] = gf.createLinearRing(list.get(i));
195             }
196             return gf.createPolygon(gf.createLinearRing(list.get(0)), holes);
197         }
198     }
199 
200     private MultiPoint createMultiPoint(List<Object> coordinates) {
201         Point[] points = new Point[coordinates.size()];
202         int i = 0;
203         for (Object obj : coordinates) {
204             @SuppressWarnings("unchecked")
205             List<Object> list = (List<Object>) obj;
206             points[i] = createPoint(list);
207             i++;
208         }
209         return getGeometryFactory().createMultiPoint(points);
210     }
211 
212     private MultiLineString createMultiLineString(List<Object> coordinates) {
213         LineString[] lineStrings = new LineString[coordinates.size()];
214         int i = 0;
215         for (Object obj : coordinates) {
216             @SuppressWarnings("unchecked")
217             List<Object> list = (List<Object>) obj;
218             lineStrings[i] = createLineString(list);
219             i++;
220         }
221         return getGeometryFactory().createMultiLineString(lineStrings);
222     }
223 
224     private MultiPolygon createMultiPolygon(List<Object> coordinates) {
225         Polygon[] polygons = new Polygon[coordinates.size()];
226         int i = 0;
227         for (Object obj : coordinates) {
228             @SuppressWarnings("unchecked")
229             List<Object> list = (List<Object>) obj;
230             polygons[i] = createPolygon(list);
231             i++;
232         }
233         return getGeometryFactory().createMultiPolygon(polygons);
234     }
235 
236     private GeometryCollection createGeometryCollection(
237             List<Geometry> geometries) {
238         GeometryCollection geometryCollection = getGeometryFactory()
239                 .createGeometryCollection(
240                         geometries.toArray(new Geometry[geometries.size()]));
241         return geometryCollection;
242     }
243 
244     private List<Geometry> parseGeometries(JsonParser jp,
245             DeserializationContext ctxt)
246                     throws IOException, JsonProcessingException {
247 
248         String type = null;
249         List<Object> coordinates = new ArrayList<Object>();
250         List<Geometry> geometries = new ArrayList<Geometry>();
251 
252         JsonToken currentToken = null;
253         while ((currentToken = jp.nextValue()) != null) {
254             if (JsonToken.END_OBJECT.equals(currentToken)) {
255                 if (type != null && !coordinates.isEmpty()) {
256                     geometries.add(createGeometry(type, coordinates));
257                     type = null;
258                     coordinates = new ArrayList<Object>();
259                 }
260             } else if (JsonToken.VALUE_STRING.equals(currentToken)) {
261                 if ("type".equals(jp.getCurrentName())) {
262                     type = jp.getText();
263                 }
264             } else if (JsonToken.START_ARRAY.equals(currentToken)) {
265                 if ("coordinates".equals(jp.getCurrentName())) {
266                     parseCoordinates(0, coordinates, jp, ctxt);
267                 }
268             } else if (JsonToken.END_ARRAY.equals(currentToken)) {
269                 break;
270             }
271         }
272         return geometries;
273     }
274 
275     private Geometry createGeometry(String type, List<Object> coordinates) {
276         if ("Point".equals(type)) {
277             return createPoint(coordinates);
278         }
279         if ("LineString".equals(type)) {
280             return createLineString(coordinates);
281         }
282         if ("Polygon".equals(type)) {
283             return createPolygon(coordinates);
284         }
285         if ("MultiPoint".equals(type)) {
286             return createMultiPoint(coordinates);
287         }
288         if ("MultiLineString".equals(type)) {
289             return createMultiLineString(coordinates);
290         }
291         if ("MultiPolygon".equals(type)) {
292             return createMultiPolygon(coordinates);
293         }
294         throw new RuntimeException(
295                 "Geometry type [" + type + "] is unsupported.");
296     }
