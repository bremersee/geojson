17  package org.bremersee.geojson.model;
18  
19  import static io.swagger.v3.oas.annotations.media.Schema.RequiredMode.REQUIRED;
20  import static org.bremersee.geojson.GeoJsonConstants.BBOX;
21  import static org.bremersee.geojson.GeoJsonConstants.FEATURE;
22  import static org.bremersee.geojson.GeoJsonConstants.GEOMETRY;
23  import static org.bremersee.geojson.GeoJsonConstants.ID;
24  import static org.bremersee.geojson.GeoJsonConstants.PROPERTIES;
25  
26  import com.fasterxml.jackson.annotation.JsonCreator;
27  import com.fasterxml.jackson.annotation.JsonIgnoreProperties;
28  import com.fasterxml.jackson.annotation.JsonProperty;
29  import com.fasterxml.jackson.annotation.JsonValue;
30  import io.swagger.v3.oas.annotations.media.Schema;
31  import java.io.Serial;
32  import java.io.Serializable;
33  import lombok.Builder;
34  import lombok.EqualsAndHashCode;
35  import lombok.Getter;
36  import lombok.NoArgsConstructor;
37  import lombok.Setter;
38  import lombok.ToString;
39  import org.bremersee.geojson.GeoJsonConstants;
40  
41  /**
42   * A Feature object represents a spatially bounded thing.
43   *
44   * @author Christian Bremer
45   */
46  @Schema(description = "A Feature object represents a spatially bounded thing.")
47  @JsonIgnoreProperties(ignoreUnknown = true)
48  @Getter
49  @EqualsAndHashCode
50  @ToString
51  @NoArgsConstructor
52  public class Feature implements Serializable {
53  
54    @Serial
55    private static final long serialVersionUID = 2L;
56  
57    /**
58     * The geo json type.
59     */
60    @Schema(description = "The feature type, must be 'Feature'.", requiredMode = REQUIRED)
61    @JsonProperty(value = FEATURE, required = true)
62    private TypeEnum type = TypeEnum.FEATURE;
63  
64    /**
65     * The identifier of this feature.
66     */
67    @Setter
68    @Schema(description = "The identifier of this feature.")
69    @JsonProperty(ID)
70    private String id = null;
71  
72    /**
73     * The bounding box.
74     */
75    @Setter
76    @Schema(description = "The bounding box.")
77    @JsonProperty(BBOX)
78    private BoundingBox bbox = null;
79  
80    /**
81     * The geometry.
82     */
83    @Setter
84    @Schema(description = "The geometry.")
85    @JsonProperty(GEOMETRY)
86    private Geometry geometry = null;
87  
88    /**
89     * The feature properties.
90     */
91    @Setter
92    @Schema(description = "The feature properties.")
93    @JsonProperty(PROPERTIES)
94    private Object properties = null;
95  
96    /**
97     * Instantiates a new feature.
98     *
99     * @param id the id
100    * @param bbox the bbox
101    * @param geometry the geometry
102    * @param properties the properties
103    */
104   @Builder(toBuilder = true)
105   public Feature(
106       String id,
107       BoundingBox bbox,
108       Geometry geometry,
109       Object properties) {
110 
111     setId(id);
112     setBbox(bbox);
113     setGeometry(geometry);
114     setProperties(properties);
115   }
116 
117   /**
118    * The feature type.
119    */
120   public enum TypeEnum {
121     /**
122      * Feature type enum.
123      */
124     FEATURE(GeoJsonConstants.FEATURE);
125 
126     private final String value;
127 
128     TypeEnum(String value) {
129       this.value = value;
130     }
131 
132     @Override
133     @JsonValue
134     public String toString() {
135       return value;
136     }
137 
138     /**
139      * From value.
140      *
141      * @param value the value
142      * @return the type enum
143      */
144     @JsonCreator
145     public static TypeEnum fromValue(String value) {
146       if (GeoJsonConstants.FEATURE.equals(value)) {
147         return FEATURE;
148       }
149       throw new IllegalArgumentException(String
150           .format("Value '%s' must be '%s'.", value, GeoJsonConstants.FEATURE));
151     }
152 
153   }
154 
155 }
