74      private static LineString createLineString(Coordinate[] coordinates) {
75          CoordinateSequence points = new CoordinateArraySequence(coordinates);
76          return new LineString(points, getGeometryFactory());
77      }
78  
79      private static LineString createLineString() {
80          return createLineString(new Coordinate[] { new Coordinate(0., 0.), new Coordinate(0., 1.),
81                  new Coordinate(1., 1.), new Coordinate(1., 0.), new Coordinate(0., 0.) });
82      }
83  
84      private static MultiLineString createMultiLineString() {
85          LineString[] lineStrings = new LineString[] { createLineString(),
86                  createLineString(new Coordinate[] { new Coordinate(2., 2.), new Coordinate(2., 4.),
87                          new Coordinate(4., 4.), new Coordinate(4., 2.), new Coordinate(2., 2.) }) };
88          return new MultiLineString(lineStrings, getGeometryFactory());
89      }
90  
91      private static Point createPoint(double x, double y) {
92          CoordinateSequence coordinates = new CoordinateArraySequence(new Coordinate[] { new Coordinate(x, y) });
93          return new Point(coordinates, getGeometryFactory());
94      }
95  
96      private static Point createPoint() {
97          return createPoint(3., 6.);
98      }
99  
100     private MultiPoint createMultiPoint() {
101         return new MultiPoint(new Point[] { createPoint(), createPoint(11., 12.) }, getGeometryFactory());
102     }
103 
104     private Polygon createPolygon() {
105         return new Polygon(new LinearRing(createLineString().getCoordinateSequence(), getGeometryFactory()), null,
106                 getGeometryFactory());
107     }
108 
109     private Polygon createPolygonWithHoles() {
110         LineString ls0 = createLineString(new Coordinate[] { new Coordinate(0.1, 0.1), new Coordinate(0.1, 0.2),
111                 new Coordinate(0.2, 0.2), new Coordinate(0.2, 0.1), new Coordinate(0.1, 0.1) });
112         LineString ls1 = createLineString(new Coordinate[] { new Coordinate(0.8, 0.8), new Coordinate(0.8, 0.9),
113                 new Coordinate(0.9, 0.9), new Coordinate(0.9, 0.8), new Coordinate(0.8, 0.8) });
114 
115         LinearRing[] holes = new LinearRing[2];
116         holes[0] = new LinearRing(ls0.getCoordinateSequence(), getGeometryFactory());
117         holes[1] = new LinearRing(ls1.getCoordinateSequence(), getGeometryFactory());
118 
119         return new Polygon(new LinearRing(createLineString().getCoordinateSequence(), getGeometryFactory()), holes,
120                 getGeometryFactory());
121     }
122 
123     private MultiPolygon createMultiPolygon() {
124         Polygon[] polygons = new Polygon[] { createPolygon(), createPolygonWithHoles() };
125         return new MultiPolygon(polygons, getGeometryFactory());
126     }

128     @Test
129     public void testPoint() throws Exception {
130         Point geometry = createPoint();
131         System.out.println("Testing " + geometry.toText());
132         String jsonStr = getObjectMapper().writerWithDefaultPrettyPrinter().writeValueAsString(geometry);
133         System.out.println(jsonStr);
134         Point readGeometry = getObjectMapper().readValue(jsonStr, Point.class);
135         TestCase.assertTrue(GeometryUtils.equals(geometry, readGeometry));
136 
137         GeometryWrapper gjg = new GeometryWrapper(geometry);
138         jsonStr = getObjectMapper().writerWithDefaultPrettyPrinter().writeValueAsString(gjg);
139         System.out.println(jsonStr);
140         GeometryWrapper readGjg = getObjectMapper().readValue(jsonStr, GeometryWrapper.class);
141         TestCase.assertEquals(gjg, readGjg);
142         TestCase.assertTrue(GeometryUtils.equals(geometry, readGjg.getGeometry()));
143     }

298         f.getProperties().put("myKey", "myValue");

307 
308         System.out.println(getObjectMapper().writerWithDefaultPrettyPrinter().writeValueAsString(readF));
309 
310         TestCase.assertEquals(f, readF);

313     @Test
314     public void testGeoJsonFeatureCollection() throws Exception {
315 
316         GeometryCollection geometry = new GeometryCollection(
317                 new Geometry[] { createLineString(), createMultiLineString() }, getGeometryFactory());
318 
319         GeoJsonFeature f1 = new GeoJsonFeature();
320         f1.setGeometry(geometry);
321         f1.setBbox(GeometryUtils.getBoundingBox(geometry));
322         f1.setCrs(new GeoJsonNamedCrs("urn:ogc:def:crs:OGC:1.3:CRS84"));
323         f1.setId("101");
324         f1.getProperties().put("myKey1", "myValue1");
325 
326         GeoJsonFeature f2 = new GeoJsonFeature();
327         f2.setGeometry(createMultiPolygon());
328         f2.setBbox(GeometryUtils.getBoundingBox(geometry));
329         f2.setCrs(new GeoJsonNamedCrs("urn:ogc:def:crs:OGC:1.3:CRS84"));
330         f2.setId("102");
331         f2.getProperties().put("myKey2", "myValue2");
332 
333         ArrayList<GeoJsonFeature> features = new ArrayList<GeoJsonFeature>();
334         features.add(f1);
335         features.add(f2);
336         GeoJsonFeatureCollection fc = new GeoJsonFeatureCollection("200",
337                 new GeoJsonNamedCrs("urn:ogc:def:crs:OGC:1.3:CRS84"), features, true, null);
338 
339         fc.getProperties().put("fcKey", "fcValue");
340 
341         System.out.println("Testing " + fc);
342 
343         String json = getObjectMapper().writerWithDefaultPrettyPrinter().writeValueAsString(fc);
344 
345         System.out.println(json);
346 
347         GeoJsonFeatureCollection readFc = getObjectMapper().readValue(json, GeoJsonFeatureCollection.class);
348 
349         System.out.println(getObjectMapper().writerWithDefaultPrettyPrinter().writeValueAsString(readFc));
350 
351         TestCase.assertEquals(fc, readFc);
352     }
