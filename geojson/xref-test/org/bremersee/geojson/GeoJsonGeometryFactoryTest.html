486     softly.assertThat(GeoJsonGeometryFactory.equals(gc1, gc4)).isFalse();

490    * Gets bounding box.
491    *
492    * @param softly the softly
493    */
494   @Test
495   void getBoundingBox(SoftAssertions softly) {
496     softly.assertThat(GeoJsonGeometryFactory.getBoundingBox((Geometry) null))
497         .isNull();
498     softly.assertThat(GeoJsonGeometryFactory.getBoundingBox((Collection<? extends Geometry>) null))
499         .isNull();
500 
501     softly.assertThat(GeoJsonGeometryFactory.getBoundingBox(target.createPoint(Double.NaN, 1.)))
502         .isNull();
503     softly.assertThat(GeoJsonGeometryFactory.getBoundingBox(target.createPoint(1., Double.NaN)))
504         .isNull();
505 
506     Point p0 = target.createPoint(1., 20.);
507     Point p1 = target.createPoint(10., 2.);
508     Geometry geometry = target.createMultiPoint(List.of(p0, p1));
509     double[] bounds = GeoJsonGeometryFactory.getBoundingBox(geometry);
510     softly.assertThat(bounds)
511         .isNotNull();
512     softly.assertThat(bounds.length)
513         .isEqualTo(4);
514     // min x:
515     softly.assertThat(bounds[0])
516         .isEqualTo(1.);
517     // min y:
518     softly.assertThat(bounds[1])
519         .isEqualTo(2.);
520     // max x:
521     softly.assertThat(bounds[2])
522         .isEqualTo(10.);
523     // max y:
524     softly.assertThat(bounds[3])
525         .isEqualTo(20.);
526 
527     Coordinate coordinate = GeoJsonGeometryFactory.getSouthWest(bounds);
528     softly.assertThat(coordinate.x)
529         .isEqualTo(1.); // min x
530     softly.assertThat(coordinate.y)
531         .isEqualTo(2.); // min y
532 
533     coordinate = GeoJsonGeometryFactory.getSouthEast(bounds);
534     softly.assertThat(coordinate.x)
535         .isEqualTo(10.); // max x
536     softly.assertThat(coordinate.y)
537         .isEqualTo(2.); // min y
538 
539     coordinate = GeoJsonGeometryFactory.getNorthEast(bounds);
540     softly.assertThat(coordinate.x)
541         .isEqualTo(10.); // max x
542     softly.assertThat(coordinate.y)
543         .isEqualTo(20.); // max y
544 
545     coordinate = GeoJsonGeometryFactory.getNorthWest(bounds);

547         .isEqualTo(1.); // min x
548     softly.assertThat(coordinate.y)
549         .isEqualTo(20.); // max y
550 
551     Polygon polygon = target.getBoundingBoxAsPolygon2D(bounds);

553         .isEqualTo(5);
554     softly.assertThat(GeoJsonGeometryFactory
555             .equals(polygon, target.getBoundingBoxAsPolygon2D(geometry)))
556         .isTrue();
557 
558     p0 = target.createPoint(new Coordinate(10., 2., 3.));
559     p1 = target.createPoint(new Coordinate(1., 20., 30.));
560     bounds = GeoJsonGeometryFactory.getBoundingBox(
561         target.createMultiPoint(List.of(p0, p1)));

563     softly.assertThat(bounds.length).isEqualTo(6);
564     // min x:
565     softly.assertThat(bounds[0])
566         .isEqualTo(1.);
567     // min y:
568     softly.assertThat(bounds[1])
569         .isEqualTo(2.);
570     // min z:
571     softly.assertThat(bounds[2])
572         .isEqualTo(3.);
573     // max x:
574     softly.assertThat(bounds[3])
575         .isEqualTo(10.);
576     // max y:
577     softly.assertThat(bounds[4])
578         .isEqualTo(20.);
579     // max z:
580     softly.assertThat(bounds[5])
581         .isEqualTo(30.);
582 
583     coordinate = GeoJsonGeometryFactory.getSouthWest(bounds);
584     softly.assertThat(coordinate.x)
585         .isEqualTo(1.); // min x
586     softly.assertThat(coordinate.y)
587         .isEqualTo(2.); // min y
588 
589     coordinate = GeoJsonGeometryFactory.getSouthEast(bounds);
590     softly.assertThat(coordinate.x)
591         .isEqualTo(10.); // max x
592     softly.assertThat(coordinate.y)
593         .isEqualTo(2.); // min y
594 
595     coordinate = GeoJsonGeometryFactory.getNorthEast(bounds);
596     softly.assertThat(coordinate.x)
597         .isEqualTo(10.); // max x
598     softly.assertThat(coordinate.y)

606 
607     softly.assertThat(GeoJsonGeometryFactory.getSouthEast(new double[0]))

617   /**
618    * Create geometry from well known text.
619    *
620    * @param softly the softly
621    * @throws IOException the io exception
622    */
623   @Test
624   void createGeometryFromWellKnownText(SoftAssertions softly) throws IOException {
625     Coordinate c1 = new Coordinate(1, 0);
626 
627     String wkt = "POINT (1 0)";
628     Geometry expected = target.createPoint(c1);
629     Geometry actual = target.createGeometryFromWellKnownText(wkt);
630     softly.assertThat(GeoJsonGeometryFactory
631             .equals(actual, expected))
632         .isTrue();
633 
634     Coordinate c3 = new Coordinate(0, 1);
635     wkt = "LINESTRING (1 0, 0 1)";
636     expected = target.createLineString(List.of(c1, c3));
637     actual = target.createGeometryFromWellKnownText(wkt);
638     softly.assertThat(GeoJsonGeometryFactory
639             .equals(actual, expected))
640         .isTrue();
641 
642     Coordinate c0 = new Coordinate(0, 0);
643     Coordinate c2 = new Coordinate(1, 1);
644     LinearRing shell = target.createLinearRing(List.of(c0, c1, c2, c3, c0));
645 
646     Coordinate h0 = new Coordinate(0.25, 0.25);
647     Coordinate h1 = new Coordinate(0.75, 0.25);
648     Coordinate h2 = new Coordinate(0.75, 0.75);
649     Coordinate h3 = new Coordinate(0.25, 0.75);
650     LinearRing hole0 = target.createLinearRing(List.of(h0, h1, h2, h3, h0));
651     wkt = "POLYGON ((0 0, 1 0, 1 1, 0 1, 0 0), "
652         + "(0.25 0.25, 0.75 0.25, 0.75 0.75, 0.25 0.75, 0.25 0.25))";
653     expected = target.createPolygon(shell, List.of(hole0));
654     actual = target.createGeometryFromWellKnownText(new StringReader(wkt));
655     softly.assertThat(GeoJsonGeometryFactory
656             .equals(actual, expected))
