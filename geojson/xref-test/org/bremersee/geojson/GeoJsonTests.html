75      getObjectMapper().registerModules(
76          new Jdk8Module(),
77          new JavaTimeModule(),
78          new GeoJsonObjectMapperModule(getGeometryFactory()));
79      getObjectMapper().disable(SerializationFeature.WRITE_DATES_AS_TIMESTAMPS);
80      getObjectMapper().enable(SerializationFeature.WRITE_DATES_WITH_ZONE_ID);
81      getObjectMapper().setDateFormat(new StdDateFormat());
82      getObjectMapper().setTimeZone(TimeZone.getTimeZone("GMT"));
83      getObjectMapper().setLocale(Locale.GERMANY);

86    private static LineString createLineString(Coordinate[] coordinates) {
87      CoordinateSequence points = new CoordinateArraySequence(coordinates);
88      return new LineString(points, getGeometryFactory());

92      return createLineString(new Coordinate[]{new Coordinate(0., 0.), new Coordinate(0., 1.),

104     CoordinateSequence coordinates = new CoordinateArraySequence(
105         new Coordinate[]{new Coordinate(x, y)});
106     return new Point(coordinates, getGeometryFactory());
107   }

123   private Polygon createPolygonWithHoles() {
124     LineString ls0 = createLineString(
125         new Coordinate[]{new Coordinate(0.1, 0.1), new Coordinate(0.1, 0.2),
126             new Coordinate(0.2, 0.2), new Coordinate(0.2, 0.1), new Coordinate(0.1, 0.1)});
127     LineString ls1 = createLineString(
128         new Coordinate[]{new Coordinate(0.8, 0.8), new Coordinate(0.8, 0.9),
129             new Coordinate(0.9, 0.9), new Coordinate(0.9, 0.8), new Coordinate(0.8, 0.8)});
130 
131     LinearRing[] holes = new LinearRing[2];
132     holes[0] = new LinearRing(ls0.getCoordinateSequence(), getGeometryFactory());
133     holes[1] = new LinearRing(ls1.getCoordinateSequence(), getGeometryFactory());
134 
135     return new Polygon(
136         new LinearRing(createLineString().getCoordinateSequence(), getGeometryFactory()), holes,
137         getGeometryFactory());
138   }

148   @Test
149   void configureGeoJsonObjectMapperModule() {
150     ObjectMapper om = new ObjectMapper();
151     GeoJsonObjectMapperModule.configure(om);
152     assertTrue(om.getRegisteredModuleIds().stream()
153         .anyMatch(module -> GeoJsonObjectMapperModule.class.getName()
154             .equals(String.valueOf(module))));
155 
156     om = new ObjectMapper();
157     om.registerModule(new GeoJsonObjectMapperModule());
158     assertTrue(om.getRegisteredModuleIds().stream()
159         .anyMatch(module -> GeoJsonObjectMapperModule.class.getName()
160             .equals(String.valueOf(module))));
161 
162     om = new ObjectMapper();
163     om.registerModule(new GeoJsonObjectMapperModule(new GeometryFactory()));
164     assertTrue(om.getRegisteredModuleIds().stream()
165         .anyMatch(module -> GeoJsonObjectMapperModule.class.getName()
166             .equals(String.valueOf(module))));
167   }

355   /**
356    * Test geometry wrapper holder.
357    *
358    * @throws Exception the exception
359    */
360   @Test
361   void testGeometryWrapperHolder() throws Exception {
362     ObjectMapper objectMapper = new ObjectMapper();
363 
364     Geometry geometry = GeometryUtils.createPoint(1., 2.);
365     GeometryWrapper wrapper = new GeometryWrapper(geometry);
366     GeometryWrapperHolder holder = new GeometryWrapperHolder();
367     holder.id = "test";
368     holder.geometry = wrapper;
369     String jsonStr = objectMapper.writerWithDefaultPrettyPrinter().writeValueAsString(holder);
370     System.out.println(jsonStr);
371 
372     GeometryWrapperHolder readHolder = objectMapper
373         .readValue(jsonStr, GeometryWrapperHolder.class);
374     assertEquals(holder.geometry, readHolder.geometry);
375 
376     geometry = GeometryUtils.createLineString(null);
377     wrapper = new GeometryWrapper(geometry);
378     holder.geometry = wrapper;
379     jsonStr = objectMapper.writerWithDefaultPrettyPrinter().writeValueAsString(holder);
380     System.out.println(jsonStr);
381 
382     readHolder = objectMapper
383         .readValue(jsonStr, GeometryWrapperHolder.class);
384     assertEquals(holder.geometry.getGeometry(), readHolder.geometry.getGeometry());
385 
386     holder.geometry = null;
387     jsonStr = objectMapper.writerWithDefaultPrettyPrinter().writeValueAsString(holder);
388     System.out.println(jsonStr);
389 
390     readHolder = objectMapper
391         .readValue(jsonStr, GeometryWrapperHolder.class);
392     assertNull(readHolder.geometry);
393 
394   }

518   /**
519    * Test bounding box.
520    */
521   @Test
522   void testBoundingBox() {
523     double[] actual = GeometryUtils.getBoundingBox((Geometry) null);
524     assertNull(actual);
525 
526     Point point = createPoint(0.1, 0.2);
527     actual = GeometryUtils.getBoundingBox(point);
528     assertNotNull(actual);
529     assertEquals(4, actual.length);
530     assertArrayEquals(new double[]{0.1, 0.2, 0.1, 0.2}, actual, 0.);
531 
532     point = GeometryUtils.createPoint(new Coordinate(0.1, 0.2, 0.3));
533     actual = GeometryUtils.getBoundingBox(point);
534     assertNotNull(actual);
535     assertEquals(6, actual.length);
536     assertArrayEquals(new double[]{0.1, 0.2, 0.3, 0.1, 0.2, 0.3}, actual, 0.);
537 
538     LineString lineString = (LineString) GeometryUtils.fromWKT("LINESTRING (3 2, 0.1 5.5)");
539     actual = GeometryUtils.getBoundingBox(lineString);
540     assertNotNull(actual);
541     assertEquals(4, actual.length);
542     assertArrayEquals(new double[]{0.1, 2., 3., 5.5}, actual, 0.);
543 
544     lineString = (LineString) GeometryUtils.fromWKT("LINESTRING (3 2 4, 0.1 5.5 1)");
545     actual = GeometryUtils.getBoundingBox(lineString);
546     assertNotNull(actual);
547     assertEquals(6, actual.length);
548     assertArrayEquals(new double[]{0.1, 2., 1., 3., 5.5, 4.}, actual, 0.);
549   }
