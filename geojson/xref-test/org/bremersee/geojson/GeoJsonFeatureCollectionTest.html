17  package org.bremersee.geojson;
18  
19  import static org.assertj.core.api.Assertions.assertThat;
20  import static org.bremersee.geojson.GeoJsonConstants.FEATURE_COLLECTION;
21  
22  import java.util.Arrays;
23  import java.util.List;
24  import org.assertj.core.api.SoftAssertions;
25  import org.assertj.core.api.junit.jupiter.SoftAssertionsExtension;
26  import org.junit.jupiter.api.Test;
27  import org.junit.jupiter.api.extension.ExtendWith;
28  import org.locationtech.jts.geom.Geometry;
29  import org.locationtech.jts.geom.MultiPoint;
30  import org.locationtech.jts.geom.Point;
31  
32  /**
33   * The GeoJSON feature collection test.
34   *
35   * @author Christian Bremer
36   */
37  @ExtendWith(SoftAssertionsExtension.class)
38  class GeoJsonFeatureCollectionTest {
39  
40    private static final GeoJsonGeometryFactory factory = new GeoJsonGeometryFactory();
41  
42    /**
43     * Gets type.
44     */
45    @Test
46    void getType() {
47      GeoJsonFeatureCollection<Point, Object> model = new GeoJsonFeatureCollection<>(false);
48      assertThat(model.getType()).isEqualTo(FEATURE_COLLECTION);
49    }
50  
51    /**
52     * Gets bounding box.
53     *
54     * @param softly the softly
55     */
56    @Test
57    void getBoundingBox(SoftAssertions softly) {
58      double[] value = new double[]{1., 1., 10., 10.};
59      GeoJsonFeatureCollection<Geometry, Object> model = new GeoJsonFeatureCollection<>(value, null);
60  
61      softly.assertThat(model.getBbox()).containsExactly(value);
62  
63      softly.assertThat(model)
64          .isEqualTo(model);
65      softly.assertThat(model.hashCode())
66          .isEqualTo(model.hashCode());
67      softly.assertThat(model)
68          .isNotEqualTo(null);
69      softly.assertThat(model)
70          .isNotEqualTo(new Object());
71      softly.assertThat(model.toString())
72          .contains(Arrays.toString(value));
73  
74      MultiPoint multiPoint = factory.createMultiPoint(List.of(
75          factory.createPoint(1., 1.),
76          factory.createPoint(5, 5.)
77      ));
78      GeoJsonFeature<Geometry, Object> f0 = new GeoJsonFeature<>(null, multiPoint, false, null);
79      Point point = factory.createPoint(10., 10.);
80      GeoJsonFeature<Geometry, Object> f1 = new GeoJsonFeature<>(null, point, false, null);
81      model = new GeoJsonFeatureCollection<>(FEATURE_COLLECTION, value, List.of(f0, f1));
82      softly.assertThat(model)
83          .isEqualTo(new GeoJsonFeatureCollection<>(List.of(f0, f1), true, null));
84  
85      double[] illegal = new double[]{1., 1., 10.};
86      softly.assertThatThrownBy(() -> new GeoJsonFeatureCollection<>(illegal, null))
87          .isInstanceOf(IllegalArgumentException.class);
88    }
89  
90    /**
91     * Gets features.
92     *
93     * @param softly the softly
94     */
95    @Test
96    void getFeatures(SoftAssertions softly) {
97      Point point = factory.createPoint(10., 10.);
98      GeoJsonFeature<Geometry, Object> value = new GeoJsonFeature<>(null, point, false, null);
99      GeoJsonFeatureCollection<Geometry, Object> model = new GeoJsonFeatureCollection<>(
100         List.of(value), false);
101 
102     softly.assertThat(model.getFeatures()).containsExactly(value);
103 
104     softly.assertThat(model)
105         .isEqualTo(model);
106     softly.assertThat(model.hashCode())
107         .isEqualTo(model.hashCode());
108     softly.assertThat(model)
109         .isNotEqualTo(null);
110     softly.assertThat(model)
111         .isNotEqualTo(new Object());
112     softly.assertThat(model.toString())
113         .contains(value.toString());
114   }
115 
116   /**
117    * Add.
118    *
119    * @param softly the softly
120    */
121   @Test
122   void add(SoftAssertions softly) {
123     GeoJsonFeatureCollection<Geometry, Object> target = new GeoJsonFeatureCollection<>(
124         true);
125     Point point = factory.createPoint(10., 10.);
126     GeoJsonFeature<Geometry, Object> value = new GeoJsonFeature<>(null, point, false, null);
127     target.add(value);
128     softly.assertThat(target.getFeatures())
129         .containsExactly(value);
130   }
131 
132   /**
133    * Add all.
134    *
135    * @param softly the softly
136    */
137   @Test
138   void addAll(SoftAssertions softly) {
139     GeoJsonFeatureCollection<Geometry, Object> target = new GeoJsonFeatureCollection<>(
140         true, (o1, o2) -> o1.getId().compareToIgnoreCase(o2.getId()));
141     Point point = factory.createPoint(10., 10.);
142     GeoJsonFeature<Geometry, Object> value0 = new GeoJsonFeature<>("0", point, false, null);
143     GeoJsonFeature<Geometry, Object> value1 = new GeoJsonFeature<>("1", point, false, null);
144     target.addAll(List.of(value1, value0));
145     softly.assertThat(target.getFeatures())
146         .containsExactly(value0, value1);
147   }
148 
149 }
